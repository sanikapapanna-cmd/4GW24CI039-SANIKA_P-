class Outer {
 void display() {
 System.out.println("Display of Outer Class");
 }
 class Inner {
 void display() {
 System.out.println("Display of Inner Class");
 }
 }
}
public class InnerClassDemo {
 public static void main(String[] args) {
 Outer o = new Outer();
 o.display();
 Outer.Inner i = o.new Inner();
 i.display();
 }
}
